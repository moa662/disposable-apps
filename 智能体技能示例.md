# File Organizer Skill

## 使用场景
当用户需要整理混乱的文件夹、清理临时文件、重命名批量文件时使用。

1. **自动整理** - 按文件类型整理到对应文件夹

2. 支持撤销操作

### scripts/organize_desktop.py
```python
#!/usr/bin/env python3
"""
桌面文件整理脚本
自动将桌面文件按类型整理到对应文件夹
"""
import os
import shutil
from pathlib import Path

def organize_desktop():
    desktop = Path.home() / "Desktop"
    folders = {
        "文档": [".doc", ".docx", ".pdf", ".txt", ".md"],
        "图片": [".jpg", ".jpeg", ".png", ".gif", ".bmp"],
        "代码": [".py", ".js", ".html", ".css", ".json"],
        "压缩包": [".zip", ".rar", ".7z", ".tar", ".gz"],
        "其他": []  # 未分类文件
    }
    
    # 创建分类文件夹
    for folder in folders.keys():
        (desktop / folder).mkdir(exist_ok=True)
    
    # 整理文件
    for file in desktop.iterdir():
        if file.is_file():
            moved = False
            for folder, extensions in folders.items():
                if file.suffix.lower() in extensions:
                    shutil.move(str(file), str(desktop / folder / file.name))
                    moved = True
                    break
            
            if not moved:
                shutil.move(str(file), str(desktop / "其他" / file.name))
    
    print(f"桌面整理完成！共整理 {len(list(desktop.iterdir()))} 个文件")

if __name__ == "__main__":
    organize_desktop()
```

### scripts/analyze_csv.py
```python
#!/usr/bin/env python3
"""
CSV数据分析脚本
"""
import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from datetime import datetime

def analyze_csv(input_file, output_file="report.html"):
    # 读取数据
    df = pd.read_csv(input_file)
    
    # 基本统计信息
    stats = {
        "数据行数": len(df),
        "数据列数": len(df.columns),
        "数据类型": df.dtypes.to_dict(),
        "缺失值统计": df.isnull().sum().to_dict(),
        "描述统计": df.describe().to_dict()
    }
    
    # 生成HTML报告
    html_content = f"""
    <!DOCTYPE html>
    <html>
    <head>
        <title>数据分析报告 - {datetime.now().strftime('%Y-%m-%d')}</title>
        <style>
            body {{ font-family: Arial, sans-serif; margin: 40px; }}
            .card {{ background: #f5f5f5; padding: 20px; margin: 20px 0; border-radius: 8px; }}
            h1 {{ color: #333; }}
            h2 {{ color: #555; }}
            table {{ border-collapse: collapse; width: 100%; }}
            th, td {{ border: 1px solid #ddd; padding: 12px; text-align: left; }}
            th {{ background-color: #4CAF50; color: white; }}
        </style>
    </head>
    <body>
        <h1>📊 数据分析报告</h1>
        <div class="card">
            <h2>📋 文件信息</h2>
            <p><strong>文件名：</strong>{input_file}</p>
            <p><strong>分析时间：</strong>{datetime.now().strftime('%Y-%m-%d %H:%M:%S')}</p>
            <p><strong>数据规模：</strong>{stats['数据行数']} 行 × {stats['数据列数']} 列</p>
        </div>
        
        <div class="card">
            <h2>📈 数据概览</h2>
            <table>
                <tr><th>字段名</th><th>数据类型</th><th>缺失值</th></tr>
    """
    
    for col, dtype in stats["数据类型"].items():
        missing = stats["缺失值统计"].get(col, 0)
        html_content += f'<tr><td>{col}</td><td>{dtype}</td><td>{missing}</td></tr>'
    
    html_content += """
            </table>
        </div>
        
        <div class="card">
            <h2>📊 统计摘要</h2>
            <p>详细统计信息已计算完成，包含均值、标准差、最小值、25%、50%、75%、最大值等指标。</p>
        </div>
    </body>
    </html>
    """
    
    with open(output_file, "w", encoding="utf-8") as f:
        f.write(html_content)
    
    print(f"✅ 分析完成！报告已保存到: {output_file}")
    return stats

if __name__ == "__main__":
    import argparse
    parser = argparse.ArgumentParser(description="CSV数据分析工具")
    parser.add_argument("input", help="输入CSV文件路径")
    parser.add_argument("--output", default="report.html", help="输出HTML报告路径")
    args = parser.parse_args()
    
    analyze_csv(args.input, args.output)
```

### scripts/create_project.py
```python
#!/usr/bin/env python3
"""
项目管理脚本 - 创建新项目
"""
import json
import os
from datetime import datetime
from pathlib import Path

def create_project(project_name, description="", team_members=None):
    """创建新项目"""
    projects_dir = Path.home() / ".workbuddy" / "projects"
    projects_dir.mkdir(parents=True, exist_ok=True)
    
    project_data = {
        "name": project_name,
        "description": description,
        "created_at": datetime.now().isoformat(),
        "status": "planning",
        "team_members": team_members or [],
        "tasks": [],
        "milestones": [],
        "documents": []
    }
    
    project_file = projects_dir / f"{project_name.replace(' ', '_')}.json"
    
    with open(project_file, "w", encoding="utf-8") as f:
        json.dump(project_data, f, ensure_ascii=False, indent=2)
    
    print(f"✅ 项目 '{project_name}' 创建成功！")
    print(f"📁 项目文件: {project_file}")
    
    # 创建项目文件夹结构
    project_folder = Path.cwd() / project_name.replace(" ", "_")
    project_folder.mkdir(exist_ok=True)
    (project_folder / "docs").mkdir(exist_ok=True)
    (project_folder / "tasks").mkdir(exist_ok=True)
    (project_folder / "reports").mkdir(exist_ok=True)
    
    return project_file

if __name__ == "__main__":
    import argparse
    parser = argparse.ArgumentParser(description="创建新项目")
    parser.add_argument("name", help="项目名称")
    parser.add_argument("--desc", help="项目描述", default="")
    parser.add_argument("--team", help="团队成员（逗号分隔）", default="")
    args = parser.parse_args()
    
    team_list = [member.strip() for member in args.team.split(",")] if args.team else []
    create_project(args.name, args.desc, team_list)
```

## 🎯 技能创建最佳实践

### 1. 技能设计原则
- **单一职责**：每个技能只专注于一个领域
- **易用性**：提供简单的使用示例
- **完整性**：包含必要的脚本和参考资料
- **安全性**：操作前备份，支持撤销

### 2. 文件结构规范
```
skill-name/
├── SKILL.md                    # 技能定义（必须）
├── scripts/                    # 可执行脚本
│   ├── main.py                # 主脚本
│   ├── utils.py               # 工具函数
│   └── install.sh             # 安装脚本（可选）
├── references/                 # 参考资料
│   ├── workflow.md            # 工作流程
│   ├── examples.md            # 使用示例
│   └── api-reference.md       # API文档
├── assets/                     # 资源文件
│   ├── templates/             # 模板文件
│   ├── configs/               # 配置文件
│   └── samples/               # 示例数据
└── README.md                   # 详细说明（可选）
```

### 3. SKILL.md 编写指南
- **前导码必须**：包含name和description
- **使用场景明确**：说明何时使用本技能
- **功能列表清晰**：使用序号列表
- **示例代码完整**：提供可运行的示例
- **注意事项全面**：包括限制和警告

### 4. 测试和验证
1. 在本地环境中测试所有脚本
2. 确保错误处理完善
3. 验证输出结果符合预期
4. 测试边界条件和异常情况
